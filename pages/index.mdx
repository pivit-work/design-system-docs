export const basePath = process.env.NEXT_PUBLIC_BASE_PATH || '';

export const DownloadIcon = () => (
  <svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round">
    <path d="M21 15v4a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2v-4" />
    <polyline points="7 10 12 15 17 10" />
    <line x1="12" y1="15" x2="12" y2="3" />
  </svg>
);

export const AssetCard = ({ src, alt, dark, downloadName, imgWidth = '50%' }) => (
  <div style={{
    flex: 1,
    minWidth: 240,
    background: dark ? '#000' : '#f2f4f7',
    borderRadius: 12,
    display: 'flex',
    flexDirection: 'column',
    alignItems: 'center',
    justifyContent: 'center',
    padding: '48px 32px 16px',
    gap: 16,
  }}>
    <img src={`${basePath}/${src}`} alt={alt} style={{ width: imgWidth, maxWidth: 200 }} />
    <button
      onClick={() => {
        fetch(`${basePath}/${src}`).then(r => r.blob()).then(blob => {
          const a = document.createElement('a');
          a.href = URL.createObjectURL(blob);
          a.download = downloadName || src;
          a.click();
          URL.revokeObjectURL(a.href);
        });
      }}
      style={{
        alignSelf: 'flex-end',
        color: dark ? '#98a2b3' : '#667085',
        cursor: 'pointer',
        padding: 8,
        borderRadius: 6,
        display: 'inline-flex',
        transition: 'opacity 0.15s',
        background: 'none',
        border: 'none',
      }}
      title="Download PNG"
    >
      <DownloadIcon />
    </button>
  </div>
);
